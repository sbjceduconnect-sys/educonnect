\subsection*{Appendix C: Installation and Setup Guide}
\addcontentsline{toc}{subsection}{Appendix C: Installation and Setup Guide}

\subsubsection*{Prerequisites}
\begin{itemize}[leftmargin=2em]
    \item Node.js $\geq$ 18
    \item Python $\geq$ 3.11
    \item Git
\end{itemize}

\subsubsection*{Backend Setup}
\begin{lstlisting}[language=bash, caption={Backend Setup Commands}]
# 1. Activate virtual environment
.\venv\Scripts\activate          # Windows
source venv/bin/activate         # Linux/Mac

# 2. Configure environment
cp educonnect-backend/.env.example educonnect-backend/.env
# Edit .env with your settings

# 3. Install dependencies
cd educonnect-backend
pip install -r requirements.txt

# 4. Run database migrations
python manage.py makemigrations
python manage.py migrate

# 5. Create admin superuser
python manage.py createsuperuser

# 6. Start development server
python manage.py runserver
# Server starts at http://localhost:8000
\end{lstlisting}

\subsubsection*{Frontend Setup}
\begin{lstlisting}[language=bash, caption={Frontend Setup Commands}]
# 1. Navigate to frontend directory
cd educonnect-client

# 2. Install dependencies
npm install

# 3. Configure environment
# Create .env with VITE_API_URL

# 4. Start development server
npm run dev
# Server starts at http://localhost:5173

# 5. Build for production
npm run build
\end{lstlisting}

\subsubsection*{Android APK Build}
\begin{lstlisting}[language=bash, caption={Android Build Commands}]
# 1. Build production frontend
npm run build

# 2. Copy to Android project
npx cap copy android

# 3. Open in Android Studio
npx cap open android

# 4. Build signed APK in Android Studio
# Build > Generate Signed Bundle/APK
\end{lstlisting}
